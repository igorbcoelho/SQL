SELECT * FROM financas.classificacoesfinanceiras
WHERE codigo IN (SELECT codigo from financas.classificacoesfinanceiras 
GROUP BY codigo
HAVING COUNT(codigo) > 1)
ORDER BY codigo


SELECT * FROM ns.pessoas
WHERE pessoa IN (SELECT pessoa from ns.pessoas 
GROUP BY pessoa
HAVING COUNT(pessoa) > 1)
ORDER BY pessoa

SELECT ID, BLOQUEADO, PESSOA, NOME, NOMEFANTASIA, CHAVECNPJ, CLIENTEATIVADO, FORNECEDORATIVADO 
FROM NS.PESSOAS 
WHERE (CLIENTEATIVADO = 1 OR LEADATIVADO = 1) AND CHAVECNPJ IS NOT NULL AND
CHAVECNPJ IN (
	SELECT CHAVECNPJ FROM NS.PESSOAS 
	WHERE (CLIENTEATIVADO = 1 OR LEADATIVADO = 1) AND CHAVECNPJ IS NOT NULL
	GROUP BY CHAVECNPJ
	HAVING COUNT(CHAVECNPJ) > 1)
ORDER BY CHAVECNPJ, PESSOA
