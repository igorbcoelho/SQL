-- ACERTAR VALOR DOS SERVICOS E VALOR DO RPS
UPDATE NS.DF_DOCFIS D SET SERVICOS = 
(SELECT SUM(S.VALOR) FROM NS.DF_SERVICOS S WHERE S.ID_DOCFIS = D.ID) 
WHERE D.EMISSAO >= '2016-07-11' AND D.ORIGEM = 3;
UPDATE NS.DF_DOCFIS SET VALOR = SERVICOS - DESCONTO WHERE EMISSAO >= '2016-07-11' AND ORIGEM = 3;
