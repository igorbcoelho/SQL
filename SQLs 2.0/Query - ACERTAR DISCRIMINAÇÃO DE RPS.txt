-- ACERTAR DISCRIMINAÇÃO DE RPS OU PRE DE ACORDO COM A CONFIGURAÇÃO DA RPS
-- POR NOTA
UPDATE NS.DF_DOCFIS SET DISCRIMINACAOSERVICOS = NS.MONTADISCRIMINACAO_RPS(DF.ID)
WHERE DF.ID = 'XXXXXXXXXXXXXXXXXXXXXX';

-- POR PERÍODO
UPDATE NS.DF_DOCFIS SET DISCRIMINACAOSERVICOS = NS.MONTADISCRIMINACAO_RPS(DF.ID)
WHERE DF.EMISSAO BETWEEN '2016-08-01' AND '2016-08-30';

-- POR MODELO, SITUACAO E PERÍODO
UPDATE NS.DF_DOCFIS SET DISCRIMINACAOSERVICOS = NS.MONTADISCRIMINACAO_RPS(DF.ID)
WHERE DF.MODELO = 'PRE' OR (DF.MODELO = 'RPS' AND DF.ORIGEM = 3 AND DF.EMISSAO >= '2016-07-20');
