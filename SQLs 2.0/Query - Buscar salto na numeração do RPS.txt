--BUSCAR SALTO NA NUMERAÇÃO
SELECT A.RPS::INTEGER -1 FROM NS.DF_DOCFIS A 
WHERE EMISSAO = '15/09/2016' 
AND A.SINAL = 0 
AND A.MODELO <> 'PRE' 
AND A.ID_ESTABELECIMENTO = '13D8744E-78B7-4290-9DA0-3B967849C214'
AND A.RPS::INTEGER <> 1 
AND NOT EXISTS (SELECT RPS::INTEGER FROM NS.DF_DOCFIS B
WHERE B.RPS::INTEGER = (A.RPS::INTEGER-1) 
AND SINAL = 0 
AND MODELO <> 'PRE' 
AND ID_ESTABELECIMENTO = '13D8744E-78B7-4290-9DA0-3B967849C214')
