UPDATE ERP.ITENSCOBRANCASPROCESSADAS SET RPS = NULL WHERE ITEMCOBRANCAPROCESSADA IN (
SELECT I.ITEMCOBRANCAPROCESSADA
  FROM ERP.TITULOSRECEBER TR
--INNER JOIN ERP.BAIXASRECEBER B ON (B.TITULORECEBER = TR.TITULORECEBER)
  INNER JOIN ERP.ITENSCOBRANCASPROCESSADAS I ON (TR.TITULORECEBER = I.TITULORECEBER)
  INNER JOIN ERP.RPSS R ON (I.RPS = R.RPS)
  WHERE TR.NUMERO IN ('1095090', '1095119', '1073732', '1092007')
--    AND TR.RPS IS NULL
  ORDER BY TR.TITULORECEBER
  );
